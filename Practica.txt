//Promedio de Calificaciones
//Guardar en un arreglo las notas de 5 estudiantes
//Calcular el promedio usando Bucle

fun main() {
    // Arreglo definido con 5 notas
    val notas = arrayOf(8, 7, 9, 6, 10)

    // Arreglo nulo para almacenar las notas
    val notasNull: Array<Int?> = arrayOfNulls(5)

    // Asignar las notas al arreglo nulo usando un ciclo
    for (i in notas.indices) {
        notasNull[i] = notas[i]
        println("√çndice: $i, Nota = ${notasNull[i]}")
    }

    // Calcular el promedio de las notas
    var suma = 0
    for (i in notas.indices) {
        suma += notasNull[i] ?: 0  // si fuera null, suma 0
    }
    val promedio = suma.toDouble() / notas.size

    println("El promedio de las notas es: $promedio")
}
